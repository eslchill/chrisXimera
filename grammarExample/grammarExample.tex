\outcome{Distinguish between simple past tense and other tenses.}

\title{Grammar Example}
\begin{document}
\begin{abstract}
The simple past tense is used for completed actions in the past.
\end{abstract}
\maketitle

\begin{question}
Jim went to the store yesterday.

\begin{solution}
\begin{hint}
This action is complete.
\end{hint}
\begin{hint}
This action occurred yesterday.
\end{hint}
\begin{hint}
Yesterday is in the past.
\end{hint}
What tense is the verb in this sentence?\answer{past}
\end{solution}
\end{question}

\begin{question}
Bart ate a sandwich for lunch.
\begin{solution}
What tense is the verb in this sentence?
\begin{multiple-choice}
\choice[correct]{simple past}
\choice{simple present}
\choice{simple future}
\choice{present perfect}
\choice{past progressive}
\end{multiple-choice}
\begin{hint}
Lunch is complete.
\end{hint}
\begin{hint}
Lunch is in the past.
\end{hint}
\end{solution}
\end{question}
